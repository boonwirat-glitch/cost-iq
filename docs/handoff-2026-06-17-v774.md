## State ของ _qnrrCompute — สิ่งที่ instance ถัดไปต้องรู้

```js
// _data structure (return จาก _qnrrCompute)
{
  quarter:        '2026q2',
  base_month:     '2026-03',
  months:         ['2026-04','2026-05','2026-06'],
  base_gmv:       13869923,    // raw sum — ใช้สำหรับ bar height calculation เท่านั้น
  base_norm:      447416.87,   // normalized daily rate — ใช้หาร NRR%
  cohort_outlets: 476,         // unique outlets ใน Mar cohort (Ning example)
  by_month: {
    '2026-04': {
      nrr_pct:   94,
      total_gmv: 13050000,
      curr_days: 30,           // เพิ่งเพิ่มใน v774
      segments:  { core_nrr, core_nrr_churn, handover, ... },
      outlets:   { core_nrr: 436, core_nrr_churn: 40, ... },
      rows:      [ ...raw CSV rows ]
    }
  }
}

// DISPLAY_BASE (ค่าที่แสดงให้ user) = Math.round(base_norm * 30)
// ≠ base_gmv (raw) — ห้ามใช้ base_gmv ใน label
```
